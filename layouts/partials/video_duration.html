{{- /* Formats an integer number of seconds as M:SS or H:MM:SS. Usage: {{ partial "video_duration.html" .Params.external.duration }} */ -}} {{- $dur := . | int -}} {{- $h := div $dur 3600 -}} {{- $m := div (mod $dur 3600) 60 -}} {{- $s := mod $dur 60 -}} {{- if gt $h 0 -}} {{- printf "%d:%02d:%02d" $h $m $s -}} {{- else -}} {{- printf "%d:%02d" $m $s -}} {{- end -}}
